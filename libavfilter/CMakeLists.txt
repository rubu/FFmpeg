    add_subdirectory(x86)
    if (APPLE)
        enable_language(ASM_NASM)
        set_source_files_properties(${AVFILTER_X86_ASSEMBLY_SOURCES} PROPERTIES LANGUAGE ASM_NASM)
        set_source_files_properties(${AVFILTER_X86_ASSEMBLY_SOURCES} PROPERTIES XCODE_EXPLICIT_FILE_TYPE "sourcecode.nasm")
    endif ()
endif ()


source_group("Assembly Files" FILES ${AVFILTER_X86_ASSEMBLY_SOURCES})
if (WIN32)
    set_source_files_properties(${AVFILTER_X86_ASSEMBLY_SOURCES} PROPERTIES VS_TOOL_OVERRIDE "NASM")
else ()
    set_source_files_properties(${AVFILTER_X86_ASSEMBLY_SOURCES} PROPERTIES LANGUAGE ASM_NASM)
endif ()
source_group("Source Files\\x86" FILES ${AVFILTER_X86_SOURCES})
add_library(avfilter ${LIBRARY_TYPE} ${AVFILTER_SOURCES} ${AVFILTER_X86_SOURCES} ${AVFILTER_X86_ASSEMBLY_SOURCES} ${AVFILTER_HEADERS} ${GENERATED_CONFIG_HEADERS} ${GENERATED_VERSION_HEADERS})
if (WIN32)
    add_custom_command(TARGET avfilter PRE_BUILD COMMAND if not exist  "$(IntDir)\\x86" mkdir "$(IntDir)\\x86")
endif ()
target_link_libraries(avfilter PUBLIC avutil avcodec avformat avresample postproc swscale ${LIBRARIES})
if (APPLE)
target_link_libraries(avfilter PUBLIC "-framework CoreImage -framework CoreGraphics -framework OpenGL -framework AppKit")
endif ()
target_include_directories(avfilter PRIVATE ${CMAKE_SOURCE_DIR} ${INCLUDES})
target_compile_definitions(avfilter PUBLIC HAVE_AV_CONFIG_H)

set_target_properties(avfilter PROPERTIES PUBLIC_HEADER "${AVFILTER_HEADERS}")
set_target_properties(avfilter PROPERTIES RUNTIME_OUTPUT_NAME "avfilter-${LIBAVFILTER_VERSION}")
install(TARGETS avfilter
    ARCHIVE
    LIBRARY
    PUBLIC_HEADER
        DESTINATION include/libavfilter
    RUNTIME
)

    install(FILES $<TARGET_PDB_FILE:avfilter> DESTINATION bin OPTIONAL)
