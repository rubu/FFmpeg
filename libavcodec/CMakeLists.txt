list(FIND TARGET_ARCHITECTURE_NAMES "X86" IS_X86)
if (NOT IS_X86 EQUAL -1)
    add_subdirectory(x86)
endif ()

source_group("Assembly Files" FILES ${AVCODEC_X86_ASSEMBLY_SOURCES})
if (WIN32)
    set_source_files_properties(${AVCODEC_X86_ASSEMBLY_SOURCES} PROPERTIES VS_TOOL_OVERRIDE "NASM")
else ()
    set_source_files_properties(${AVCODEC_X86_ASSEMBLY_SOURCES} PROPERTIES LANGUAGE ASM_NASM)
endif ()
source_group("Source Files\\x86" FILES ${AVCODEC_X86_SOURCES})
add_library(avcodec ${LIBRARY_TYPE} ${AVCODEC_SOURCES} ${AVCODEC_X86_SOURCES} ${AVCODEC_X86_ASSEMBLY_SOURCES} ${AVCODEC_HEADERS} ${GENERATED_CONFIG_HEADERS} ${GENERATED_VERSION_HEADERS} ${EXPORT_DEFINITIONS})
target_link_libraries(avcodec PUBLIC avutil swresample ${EXTERNAL_LIBRARIES})
if (WIN32)
    add_custom_command(TARGET avcodec PRE_BUILD COMMAND if not exist  "$(IntDir)\\x86" mkdir "$(IntDir)\\x86")
endif ()
target_include_directories(avcodec PRIVATE ${CMAKE_SOURCE_DIR} ${EXTERNAL_INCLUDE_DIRECTORIES})
target_compile_definitions(avcodec PUBLIC HAVE_AV_CONFIG_H)
set_target_properties(avcodec PROPERTIES PUBLIC_HEADER "${AVCODEC_HEADERS}")
set_target_properties(avcodec PROPERTIES RUNTIME_OUTPUT_NAME "avcodec-${LIBAVCODEC_VERSION}")
install(TARGETS avcodec
    ARCHIVE
    LIBRARY
    PUBLIC_HEADER
        DESTINATION include/libavcodec
    RUNTIME
)
if (WIN32 AND "${LIBRARY_TYPE}" STREQUAL "SHARED")
    install(FILES $<TARGET_PDB_FILE:avcodec> DESTINATION bin OPTIONAL)
endif ()
